9 ! 1 - This subroutine creates or replaces (if it already exists) the output. Not using the correct status creates errors.
10
11 subroutine create_output(u,name_file)
12 implicit none
13 integer, intent(in) :: u
14 character(100), intent(in) :: name_file
15 integer :: ios
16 logical :: exists
17
18 inquire(file=name_file, exist=exists)
19 if (exists) then
20 ! if the file already exists we use the status REPLACE, this way we overwrite the alredy existing file.
21 open(unit=u, file=name_file, status='REPLACE', action='WRITE', iostat=ios)
22 else
23 ! if not, we use the status NEW, this way we write on a freshly created file.
24
25 open(unit=u, file=name_file, status='NEW', action='WRITE', iostat=ios)
26 end if
27
28 ! Checks that the file was opened correctly using the ios = 0.
29 if (ios /= 0) then
30 write(10,'(A,I0)') 'Error opening', name_file,' IOSTAT = ', ios
31 stop 1
32 end if
